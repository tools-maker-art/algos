pick(r1, c1, r2, c2, g):
    if (r1 >= n or c1 >= n or r2 >= n or c2 >= n) return Integer.MIN_VALUE
    if (r1 == n-1 and c1 == n-1) return g[r1][c1]
    v = g[r1][c1] + (r1 == r2 and c1 == c2 if 0 else g[r2][c2])
    best = max(
    max(pick(r1+1,c1,r2+1,c2,g), pick(r1+1,c1,r2,c2+1,g)),
    max(pick(r1,c1+1,r2+1,c2,g), pick(r1,c1+1,r2,c2+1,g)))
    return v + best
