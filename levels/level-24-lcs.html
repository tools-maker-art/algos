what is the recurrence?

The DP recurrence in plain math:

int lcs(int i, int j, String a, String b) {
    if (i == 0 || j == 0) return 0;
    if (a.charAt(i-1) == b.charAt(j-1))
        return 1 + lcs(i-1, j-1, a, b);
    return Math.max(lcs(i-1, j, a, b), lcs(i, j-1, a, b));
}

Integer[][] memo;
int lcs(int i, int j, String a, String b) {
    if (i == 0 || j == 0) return 0;
    if (memo[i][j] != null) return memo[i][j];
    if (a.charAt(i-1) == b.charAt(j-1))
        return memo[i][j] = 1 + lcs(i-1, j-1, a, b);
    return memo[i][j] = Math.max(lcs(i-1, j, a, b), lcs(i, j-1, a, b));
}

int[][] dp = new int[n+1][m+1];
for (int i = 1; i <= n; i++)
    for (int j = 1; j <= m; j++) {
        if (a.charAt(i-1) == b.charAt(j-1))
            dp[i][j] = 1 + dp[i-1][j-1];
        else
            dp[i][j] = Math.max(dp[i-1][j], dp[i][j-1]);
    }
return dp[n][m];

JAVA   space-optimised DP ⬆ tiny memory 
int[] prev = new int[m+1], curr = new int[m+1];
for (int i = 1; i <= n; i++) {
    for (int j = 1; j <= m; j++) {
        if (a.charAt(i-1) == b.charAt(j-1))
            curr[j] = 1 + prev[j-1];
        else
            curr[j] = Math.max(prev[j], curr[j-1]);
    }
    int[] t = prev; prev = curr; curr = t;
}
return prev[m];
