rob(m):
    if (m.length == 1) return m[0]
    return max(robLine(m, 0, m.length-2),
    robLine(m, 1, m.length-1))

robLine(m, l, r):
    p2 = 0, p1 = 0
    for i from l to r:

    c = max(p1, p2 + m[i])
    p2 = p1; p1 = c

return p1
