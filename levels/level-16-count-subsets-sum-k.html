int cnt(int i, int k, int[] nums) {
    if (k == 0) return 1;
    if (i < 0) return 0;
    int skip = cnt(i-1, k, nums);
    int pick = (k >= nums[i]) ? cnt(i-1, k-nums[i], nums) : 0;
    return skip + pick;
}

int[] dp = new int[K+1];
dp[0] = 1;
for (int x : nums)
    for (int k = K; k >= x; k--)
        dp[k] += dp[k - x];
return dp[K];
