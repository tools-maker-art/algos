int total = 0; for (int x : nums) total += x;
boolean[] dp = new boolean[total+1];
dp[0] = true;
for (int x : nums)
    for (int k = total; k >= x; k--)
        dp[k] = dp[k] || dp[k - x];
int best = Integer.MAX_VALUE;
for (int k = 0; k <= total/2; k++)
    if (dp[k]) best = Math.min(best, total - 2*k);
return best;
