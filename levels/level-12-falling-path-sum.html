❄️ The story

A snowflake falls down a grid. Each row, it can drop straight or shift one column.

what is the recurrence?

The DP recurrence in plain math:

dp(i,j) = grid[i][j] + min(dp(i-1,j-1), dp(i-1,j), dp(i-1,j+1))

int fall(int i, int j, int[][] g) {
    if (j < 0 || j >= g[0].length) return Integer.MAX_VALUE;
    if (i == 0) return g[0][j];
    return g[i][j] + Math.min(fall(i-1,j-1,g),
                Math.min(fall(i-1,j,g), fall(i-1,j+1,g)));
}

Integer[][] memo;
int fall(int i, int j, int[][] g) {
    if (j < 0 || j >= g[0].length) return Integer.MAX_VALUE;
    if (i == 0) return g[0][j];
    if (memo[i][j] != null) return memo[i][j];
    return memo[i][j] = g[i][j] + Math.min(fall(i-1,j-1,g),
        Math.min(fall(i-1,j,g), fall(i-1,j+1,g)));
}

int[][] dp = new int[m][n];
dp[0] = g[0].clone();
for (int i = 1; i < m; i++)
    for (int j = 0; j < n; j++) {
        int a = dp[i-1][j];
        int b = (j > 0) ? dp[i-1][j-1] : Integer.MAX_VALUE;
        int c = (j < n-1) ? dp[i-1][j+1] : Integer.MAX_VALUE;
        dp[i][j] = g[i][j] + Math.min(a, Math.min(b, c));
    }
int best = Integer.MAX_VALUE;
for (int v : dp[m-1]) best = Math.min(best, v);
return best;
