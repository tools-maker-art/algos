int best(int i, int w, int[] wt, int[] val) {
    if (i < 0 || w == 0) return 0;
    int skip = best(i-1, w, wt, val);
    int pick = (w >= wt[i]) ? val[i] + best(i-1, w - wt[i], wt, val) : 0;
    return Math.max(skip, pick);
}

Integer[][] memo;
int best(int i, int w, int[] wt, int[] val) {
    if (i < 0 || w == 0) return 0;
    if (memo[i][w] != null) return memo[i][w];
    int skip = best(i-1, w, wt, val);
    int pick = (w >= wt[i]) ? val[i] + best(i-1, w - wt[i], wt, val) : 0;
    return memo[i][w] = Math.max(skip, pick);
}

int[][] dp = new int[n+1][W+1];
for (int i = 1; i <= n; i++)
    for (int w = 0; w <= W; w++) {
        dp[i][w] = dp[i-1][w];
        if (w >= wt[i-1])
            dp[i][w] = Math.max(dp[i][w], val[i-1] + dp[i-1][w - wt[i-1]]);
    }
return dp[n][W];
