boolean can(int i, int k, int[] nums) {
    if (k == 0) return true;
    if (i < 0) return false;
    boolean skip = can(i-1, k, nums);
    boolean pick = (k >= nums[i]) && can(i-1, k - nums[i], nums);
    return skip || pick;
}

boolean[] dp = new boolean[K+1];
dp[0] = true;
for (int x : nums)
    for (int k = K; k >= x; k--)
        dp[k] = dp[k] || dp[k - x];
return dp[K];
